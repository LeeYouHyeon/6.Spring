-- 2025-05-30
create table board(
bno bigint auto_increment,
title varchar(200) not null,
writer varchar(200) not null,
content text,
is_del varchar(5) default 'N',
reg_date datetime default now(),
read_count int default 0,
primary key(bno)
);
